### TripAccumulator and TripSnapshot

```dart
// lib/features/tracking/services/trip_accumulator.dart

/// Instantaneous read-only view of accumulator state, sent from the service
/// isolate to the UI isolate via flutter_background_service.invoke every
/// kTrackingUiUpdateInterval.
class TripSnapshot {
  const TripSnapshot({
    required this.startedAt,
    required this.elapsedSeconds,
    required this.distanceMeters,
    required this.timeMovingSeconds,
    required this.timeStuckSeconds,
    required this.currentSpeedMs,
  });

  final DateTime startedAt;
  final int elapsedSeconds;
  final double distanceMeters;
  final int timeMovingSeconds;
  final int timeStuckSeconds;
  final double currentSpeedMs;

  Map<String, Object?> toMap();
  static TripSnapshot fromMap(Map<String, Object?> map);
}

/// Streaming accumulator for a single in-progress trip. Owned by the
/// service isolate (plan 02-03). Lifecycle:
///   1. Construct with `startedAt` (wall-clock UTC).
///   2. Call `addSample(position)` for every accepted GPS fix.
///   3. Call `snapshot(now)` on the 1-Hz UI timer.
///   4. Call `finalize(endedAt)` once on Stop.
///
/// Important: interval classification uses `prev.speed >= kStuckSpeedThresholdMs`
/// (D-03). Using `Position.speed` in m/s directly against `kStuckSpeedThresholdMs`
/// avoids Pitfall 2's per-sample unit conversion. Accuracy > kTrackingMax* is
/// dropped. Time gaps > kTrackingMaxAttributableGapSeconds are recorded in the
/// sample list but NOT attributed to the moving/stuck buckets.
class TripAccumulator {
  TripAccumulator({required this.startedAt});

  final DateTime startedAt;

  /// Add one GPS fix. Safe to call multiple times per second. Idempotent on
  /// stale/low-quality samples (they are silently dropped).
  void addSample(Position position);

  /// Project current state into a snapshot for UI streaming.
  TripSnapshot snapshot(DateTime now);

  /// Build the final FinalizedTrip. After calling finalize, further addSample
  /// calls are ignored. Returns a DTO safe to cross isolate boundaries via
  /// toMap()/fromMap().
  FinalizedTrip finalize(DateTime endedAt);

  /// For testing: current distance accumulator in meters.
  @visibleForTesting
  double get distanceMetersForTest;

  /// For testing: current moving-seconds accumulator.
  @visibleForTesting
  int get timeMovingSecondsForTest;

  /// For testing: current stuck-seconds accumulator.
  @visibleForTesting
  int get timeStuckSecondsForTest;
}
```

  final DateTime endTime;        // UTC
  final int durationSeconds;
  final double distanceMeters;

    List<({double lat, double lng})> decodePolyline(String encoded) {
      final result = <({double lat, double lng})>[];
      var index = 0;
      var lat = 0;
      var lng = 0;
      while (index < encoded.length) {
        int shift = 0;
        int dLat = 0;
        int b;
        do {
          b = encoded.codeUnitAt(index++) - 63;
          dLat |= (b & 0x1f) << shift;
          shift += 5;
        } while (b >= 0x20);
        lat += (dLat & 1) != 0 ? ~(dLat >> 1) : dLat >> 1;

        shift = 0;
        int dLng = 0;
        do {
          b = encoded.codeUnitAt(index++) - 63;
          dLng |= (b & 0x1f) << shift;
          shift += 5;
        } while (b >= 0x20);
        lng += (dLng & 1) != 0 ? ~(dLng >> 1) : dLng >> 1;

        result.add((lat: lat / 1e5, lng: lng / 1e5));
      }
      return result;
    }
    ```

    class TripAccumulator {
      TripAccumulator({required this.startedAt})
          : _tripId = const Uuid().v4();

      final DateTime startedAt;
      final String _tripId;
      Position? _lastAccepted;
      double _distanceMeters = 0;
      int _timeMovingSeconds = 0;
      int _timeStuckSeconds = 0;
      final List<Position> _samples = <Position>[];
      bool _finalized = false;

      void addSample(Position p) {
        if (_finalized) return;
        if (p.accuracy > kTrackingMaxAcceptableAccuracyMeters) return;

        final prev = _lastAccepted;
        if (prev == null) {
          _lastAccepted = p;
          _samples.add(p);
          return;
        }

        final deltaMillis =
            p.timestamp.difference(prev.timestamp).inMilliseconds;
        if (deltaMillis <= 0) {
          _samples.add(p);
          _lastAccepted = p;
          return;
        }
        final deltaSec = deltaMillis / 1000.0;

        // Distance always accumulated (Haversine is robust to any gap size).
        _distanceMeters += Geolocator.distanceBetween(
          prev.latitude,
          prev.longitude,
          p.latitude,
          p.longitude,
        );

        // Time attribution is gap-guarded (Pitfall 2 extension).
        if (deltaSec <= kTrackingMaxAttributableGapSeconds) {
          final deltaSecInt = deltaSec.round();
          // D-03: prev.speed classifies the prev→curr INTERVAL.
          // kStuckSpeedThresholdMs is pre-converted so this comparison is
          // unit-correct without per-sample math (Pitfall 2).
          if (prev.speed >= kStuckSpeedThresholdMs) {
            _timeMovingSeconds += deltaSecInt;
          } else {
            _timeStuckSeconds += deltaSecInt;
          }
        }

        _lastAccepted = p;
        _samples.add(p);
      }

      TripSnapshot snapshot(DateTime now) {
        return TripSnapshot(
          startedAt: startedAt,
          elapsedSeconds: now.difference(startedAt).inSeconds,
          distanceMeters: _distanceMeters,
          timeMovingSeconds: _timeMovingSeconds,
          timeStuckSeconds: _timeStuckSeconds,
          currentSpeedMs: _lastAccepted?.speed ?? 0,
        );
      }

      FinalizedTrip finalize(DateTime endedAt) {
        _finalized = true;
        final encoded = encodePolyline(
          _samples
              .map((p) => (lat: p.latitude, lng: p.longitude))
              .toList(growable: false),
        );
        return FinalizedTrip(
          id: _tripId,
          startTime: startedAt.toUtc(),
          endTime: endedAt.toUtc(),
          durationSeconds: endedAt.difference(startedAt).inSeconds,
          distanceMeters: _distanceMeters,
          timeMovingSeconds: _timeMovingSeconds,
          timeStuckSeconds: _timeStuckSeconds,
          encodedPolyline: encoded,
        );
      }
    }
    ```
