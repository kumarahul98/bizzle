### Sealed TrackingState

```dart
// lib/features/tracking/state/tracking_state.dart

/// High-level state of the tracking feature as observed by the UI isolate.
/// Sealed so exhaustiveness-checked switches are possible at call sites.
sealed class TrackingState {
  const TrackingState();
}

/// No trip in progress. Shown when the app opens and after a trip is saved
/// or discarded.
final class TrackingIdle extends TrackingState {
  const TrackingIdle();
}

/// User tapped Start, the service is initialising, but the first GPS
/// sample has not yet arrived. UI shows a spinner, not the tiles.
final class TrackingStarting extends TrackingState {
  const TrackingStarting();
}

/// Live tracking — every field is refreshed at kTrackingUiUpdateInterval
/// from the service isolate.
final class TrackingActive extends TrackingState {
  const TrackingActive({
    required this.startedAt,
    required this.elapsedSeconds,
    required this.distanceMeters,
    required this.currentSpeedKmh,
    required this.timeMovingSeconds,
    required this.timeStuckSeconds,
  });

  final DateTime startedAt;
  final int elapsedSeconds;
  final double distanceMeters;
  final double currentSpeedKmh;   // UI unit — converted from m/s at notifier
  final int timeMovingSeconds;
  final int timeStuckSeconds;
}

/// Stop tapped but persistence not yet complete. Brief — UI may show a
/// "Saving trip" spinner or just dim the Stop button.
final class TrackingStopping extends TrackingState {
  const TrackingStopping();
}

/// Terminal error path. Message is user-facing.
final class TrackingError extends TrackingState {
  const TrackingError(this.message);
  final String message;
}
```

```dart
// lib/features/tracking/services/tracking_service.dart

/// flutter_background_service onStart entrypoint. Runs in a background
/// isolate. Declared top-level so it survives Dart tree-shaking.
@pragma('vm:entry-point')
Future<void> trackingServiceOnStart(ServiceInstance service);

/// Called from main() in plan 02-05 to configure FlutterBackgroundService.

```

`trackingServiceOnStart` contract (plan 02-05 will add notification show/dismiss — THIS plan emits events only):
- Create `TripAccumulator(startedAt: DateTime.now())`
- Subscribe to `Geolocator.getPositionStream(locationSettings: ...)` using kTrackingSampleInterval
- For each Position: early-return if _stopping; else `accumulator.addSample(position)`
- On `Timer.periodic(kTrackingUiUpdateInterval)`: `service.invoke('tracking_state', accumulator.snapshot(DateTime.now()).toMap())`
- On `service.on('stop_tracking').listen(...)`: set `_stopping = true`, cancel position subscription, build `FinalizedTrip`, `service.invoke('trip_finalized', trip.toMap())`, stop the UI timer, call `service.stopSelf()`.

    ```dart
    @pragma('vm:entry-point')
    Future<void> trackingServiceOnStart(ServiceInstance service) async {
      // Android-only in Phase 2 (per project constraints).
      if (service is AndroidServiceInstance) {
        // Plan 02-05 will add:
        //   await service.setForegroundNotificationInfo(...);
        // and/or dismiss the fbs stock notification in favour of our
        // flutter_local_notifications one. Left as a hook here so plan 02-05
        // has an obvious insertion point — NOT a TODO, a documented handoff.
      }

      final accumulator = TripAccumulator(startedAt: DateTime.now().toUtc());
      var stopping = false;
      StreamSubscription<Position>? positionSub;
      Timer? uiTimer;

      const settings = AndroidSettings(
        accuracy: LocationAccuracy.high,
        distanceFilter: 0,
        intervalDuration: kTrackingSampleInterval,
      );

      positionSub = Geolocator.getPositionStream(locationSettings: settings)
          .listen((position) {
        if (stopping) return;              // Race guard (RESEARCH §8).
        accumulator.addSample(position);
      });

      uiTimer = Timer.periodic(kTrackingUiUpdateInterval, (_) {
        if (stopping) return;
        service.invoke(
          kTrackingStateEvent,
          accumulator.snapshot(DateTime.now().toUtc()).toMap(),
        );
      });

      service.on(kStopTrackingEvent).listen((_) async {
        // Order matters: set the flag BEFORE cancelling the subscription so
        // any in-flight listener callback early-returns instead of touching
        // a disposed accumulator (RESEARCH §8).
        stopping = true;
        await positionSub?.cancel();
        uiTimer?.cancel();
        final trip = accumulator.finalize(DateTime.now().toUtc());
        service.invoke(kTripFinalizedEvent, trip.toMap());
        await service.stopSelf();
      });
    }

    /// Called once from main() in plan 02-05. Configures FlutterBackgroundService
    /// with the SAME Android channel + notification id that our
    /// flutter_local_notifications 'Recording commute' notification uses.
    ///
    /// D-14 UNIFICATION CONTRACT (DO NOT DEVIATE):
    /// Android dedupes notifications by (channelId, notificationId). If we let
    /// flutter_background_service spawn its stock notification with a DIFFERENT
    /// id, the user sees TWO shade entries during tracking — one stock, one
    /// ours. Instead we pin `foregroundServiceNotificationId` to the exact
    /// same `kTrackingNotificationId` that `TrackingNotificationService.showRecording()`
    /// uses in plan 02-05. Plan 02-05 then calls `_plugin.show(kTrackingNotificationId, ...)`
    /// which replaces the transient placeholder with the real Stop-action
    /// notification on the same channel + id → Android collapses them into
    /// a single entry. The initial title/content below are a brief placeholder
    /// that plan 02-05 overwrites immediately; they must still be non-empty
    /// for fbs to start the service.
    @pragma('vm:entry-point')
    Future<void> configureBackgroundService() async {
      await FlutterBackgroundService().configure(
        androidConfiguration: AndroidConfiguration(
          onStart: trackingServiceOnStart,
          autoStart: false,
          isForegroundMode: true,
          // D-14: same channel id the UX-03 notification uses (plan 02-05).
          notificationChannelId: kTrackingNotificationChannelId,
          initialNotificationTitle: kTrackingNotificationTitle,
          initialNotificationContent: '',
          // D-14: same id the UX-03 notification uses (plan 02-05). This is
          // the dedup contract — see function doc above.
          foregroundServiceNotificationId: kTrackingNotificationId,
          foregroundServiceTypes: [AndroidForegroundType.location],
        ),
        iosConfiguration: IosConfiguration(
          autoStart: false,
        ),
      );
    }
    ```

    ```dart
    class TrackingNotifier extends Notifier<TrackingState> {
      StreamSubscription<Map<String, dynamic>?>? _stateSub;
      StreamSubscription<Map<String, dynamic>?>? _finalizeSub;

      @override
      TrackingState build() {
        ref.onDispose(() {
          _stateSub?.cancel();
          _finalizeSub?.cancel();
        });
        _attach();
        return const TrackingIdle();
      }

      void _attach() {
        final service = FlutterBackgroundService();
        _stateSub = service.on(kTrackingStateEvent).listen((data) {
          if (data == null) return;
          // strict-casts compliant — helper does the shape check.
          state = trackingActiveFromSnapshotMap(
            data.cast<String, Object?>(),
          );
        });
        _finalizeSub = service.on(kTripFinalizedEvent).listen((data) {
          if (data == null) return;
          state = const TrackingStopping();
          // Plan 02-05 adds:
          //   await ref.read(trackingServiceControllerProvider)
          //       .persistFinalizedTrip(
          //         FinalizedTrip.fromMap(data.cast<String, Object?>()),
          //       );
          // and then the UI-isolate short-trip snackbar / notification dismiss.
          // In plan 02-03 we simply return to idle — unsaved — so the feature
          // can be manually smoke-tested on an emulator without persistence.
          state = const TrackingIdle();
        });
      }

      Future<void> start() async {
        if (state is TrackingActive || state is TrackingStarting) return;
        state = const TrackingStarting();
        final ok =
            await ref.read(trackingServiceControllerProvider).start();
        if (!ok) {
          state = const TrackingError('Unable to start tracking');
        }
      }

      Future<void> stop() async {
        if (state is! TrackingActive) return;
        await ref.read(trackingServiceControllerProvider).stop();
        // Actual transition to TrackingStopping / TrackingIdle happens in
        // the trip_finalized listener above.
      }
    }
    ```
